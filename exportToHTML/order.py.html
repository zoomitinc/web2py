832          if distance is None: 
833              distance = float(self.distance) 
834          else: 
835              distance = float(distance) 
836          price = float(0) 
837          if origin_agent is not None: 
838              if distance <= origin_agent.fleetInfoTransportationRates.milesInBaseRate: 
839                  price += origin_agent.fleetInfoTransportationRates.minBaseRate 
840              elif distance > origin_agent.fleetInfoTransportationRates.milesInBaseRate: 
841                  price = origin_agent.fleetInfoTransportationRates.minBaseRate + (origin_agent.fleetInfoTransportationRates.perMileRate * (distance-origin_agent.fleetInfoTransportationRates.milesInBaseRate)) 
842              rounded_price = ("%.2f" % round(price,2)) 
843              return rounded_price 
844          elif origin_agent is None: 
845              if distance <= float(20.00): 
846                  price = 30.00 
847                  return price 
848              else: 
849                  price = 30.00 + (distance * float(1.50))
